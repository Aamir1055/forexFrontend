# Place this file's contents into: C:\xampp\htdocs\.htaccess
# Purpose: Proxy /api requests to backend on :8080 to eliminate CORS

# REQUIREMENTS:
# 1) In C:\xampp\apache\conf\httpd.conf, ensure these are enabled (remove #):
#    LoadModule proxy_module modules/mod_proxy.so
#    LoadModule proxy_http_module modules/mod_proxy_http.so
#    LoadModule rewrite_module modules/mod_rewrite.so
# 2) In your DocumentRoot directory permissions, AllowOverride All must be set (default XAMPP usually allows it)
# 3) Restart Apache after creating this file

<IfModule mod_rewrite.c>
  RewriteEngine On
  
  # Proxy /api/* to backend service
  # Example: http://185.136.159.142/api/users -> http://127.0.0.1:8080/api/users
  RewriteRule ^api/(.*)$ http://127.0.0.1:8080/api/$1 [P,L]
</IfModule>
